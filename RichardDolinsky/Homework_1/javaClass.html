public class Zapisovace { public static void main(String[] args) {
// vytvor prazdne pole pre 50 cisel
int[] poleCisel = new int[50];
// napln ho 50 nahodnymi cislami
Random random = new Random();
for (int i = 0; i < 50; i++) {
poleCisel[i] = random.nextInt(100);
}
// najdi vsetky neparne cislo v poli a oddel ich do noveho pola
 ArrayList neparneCislo = new ArrayList<>();
  ArrayList parneCislo = new ArrayList<>();
for (int i = 0; i < poleCisel.length; i++) {
if (poleCisel[i] % 2 == 1) {
neparneCislo.add(poleCisel[i]);
} else {
parneCislo.add(poleCisel[i]);
}

}
System.out.println(" Parne cisla " + parneCislo.toString());
System.out.println(" Neparne cisla " + neparneCislo.toString());

}
